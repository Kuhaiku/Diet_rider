function notify(t, type = "success") {
  if (typeof Toastify === "function") {
    Toastify({
      text: t,
       duration: 2000,
    gravity: "top",
    position: "center",
    style: {
      background: type === "error" ? "#ef4444" : "#22c55e",
      borderRadius: "8px",
      },
    }).showToast();
  } else {
    alert(t);
  }
}
